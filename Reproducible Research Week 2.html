sum(is.na(data))

  1. filling the missing value. here mean of 5-min interval was assigned to corresponding missing value. new dataset with filled missing value is called “Inser2”
Inser <- cbind(data, DailyAver[,2])
NA2aver <- function(x) { 
  if (is.na(x[1])) {
    x[1] = x[4]
  }
  x}
Inser2 <- apply(Inser, 1, NA2aver)
Inser2 <- t(Inser2)
colnames(Inser2) <- c("steps", "date", "interval", "average_steps")
Inser2 <- as.data.frame(Inser2, stringsAsFactors = FALSE)
